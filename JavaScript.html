// Créer un canvas

let canvas = document.createElement("canvas");

let ctx = canvas.getContext("2d"); canvas.width = 800; canvas.height = 600; document.body.appendChild(canvas);

// Charger les images des voitures

let car1 = new Image(); car1.src = "car1.png"; let car2 = new Image(); car2.src = "car2.png";

// Définir les positions des voitures

let car1X = 100; let car2X = 500;

// Fonction pour dessiner les voitures sur le canvas

function drawCars() {

ctx.drawImage(car1, car1X, 100, 50, 100);

ctx.drawImage(car2, car2X, 400, 50, 100);
}

// Fonction pour mettre à jour la position des voitures

function updateCars() {

car1X += Math.random() * 5;

// Mouvement aléatoire

car2X += Math.random() * 5;

// Mouvement aléatoire

}

// Fonction principale pour dessiner et mettre à jour le jeu

function mainLoop() {

ctx.clearRect(0, 0, canvas.width, canvas.height);

// Effacer le canvas
drawCars();

// Dessiner les voiture
updateCars();

// Mettre à jour la position des voiture
}

// Boucle principale du jeu (appelée toutes les 10 millisecondes) setInterval(mainLoop, 10);

Ce code crée un canvas dans la page web et charge deux images de voitures. Les positions des voitures sont définies dans des variables, et une fonction `drawCars()` est utilisée pour dessiner les voitures sur le canvas. Une fonction `updateCars()` est également définie pour mettre à jour la position des voitures à chaque itération de la boucle principale du jeu. Finalement, une boucle principale est mise en place pour appeler les fonctions `drawCars()` et `updateCars()` toutes les 10 millisecondes afin de créer l'illusion de mouvement des voitures. Bien sûr, ce n'est qu'un exemple très simple et il y a beaucoup de choses que vous pouvez ajouter pour rendre le jeu plus intéressant !
